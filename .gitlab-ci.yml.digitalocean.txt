variables:
  DOCKER_DRIVER: overlay2
  DOCKER_TLS_CERTDIR: "/certs"
  KUBE_NAMESPACE: "kyc-blockchain-frontend"

  # Trivy settings
  # CRITICAL severity blocks the pipeline. HIGH is reported but non-blocking.
  # To also block on HIGH: set TRIVY_SEVERITY to "HIGH,CRITICAL"
  TRIVY_SEVERITY: "CRITICAL"
  TRIVY_EXIT_CODE: "1"            # exit 1 = fail pipeline on findings
  TRIVY_IGNORE_UNFIXED: "true"    # skip vulns with no fix yet (reduces noise)
  TRIVY_NO_PROGRESS: "true"       # cleaner CI log output
  TRIVY_FORMAT: "table"           # human-readable in CI log

stages:
  - build
  - scan
  - deploy

# ─── Trivy Security Scan ──────────────────────────────────────────────────────
# Scans the image that was just pushed.
# Three reports are produced:
#   1. Table in the CI log                    — immediate human-readable feedback
#   2. trivy-report.json                      — full report (downloadable artifact)
#   3. gl-container-scanning-report.json      — GitLab Security Dashboard format
#
# Scans the registry image directly so the exact artifact that will be deployed
# is what gets scanned — no local cache or rebuild inconsistency.
scan:
  stage: scan
  image:
    name: aquasec/trivy:latest
    entrypoint: [""]   # override so we can run shell commands
  needs:
    - job: build       # only runs after build succeeds
  before_script:
    - trivy registry login --username $CI_REGISTRY_USER --password $CI_REGISTRY_PASSWORD $CI_REGISTRY
  script:
    - echo "Running Trivy security scan on $CI_REGISTRY_IMAGE:$CI_COMMIT_SHA ..."
 
    # ── 1. Table report in CI log (blocking on CRITICAL) ──────────────────
    - |
      trivy image \
        --severity $TRIVY_SEVERITY \
        --exit-code $TRIVY_EXIT_CODE \
        --ignore-unfixed=$TRIVY_IGNORE_UNFIXED \
        --no-progress=$TRIVY_NO_PROGRESS \
        --format $TRIVY_FORMAT \
        $CI_REGISTRY_IMAGE:$CI_COMMIT_SHA
 
    # ── 2. Full JSON report (artifact — HIGH + CRITICAL, non-blocking) ────
    - |
      trivy image \
        --severity HIGH,CRITICAL \
        --exit-code 0 \
        --ignore-unfixed=$TRIVY_IGNORE_UNFIXED \
        --no-progress=$TRIVY_NO_PROGRESS \
        --format json \
        --output trivy-report.json \
        $CI_REGISTRY_IMAGE:$CI_COMMIT_SHA
 
    # ── 3. GitLab Security Dashboard format ──────────────────────────────
    - |
      trivy image \
        --severity HIGH,CRITICAL \
        --exit-code 0 \
        --ignore-unfixed=$TRIVY_IGNORE_UNFIXED \
        --no-progress=$TRIVY_NO_PROGRESS \
        --format template \
        --template "@/contrib/gitlab.tpl" \
        --output gl-container-scanning-report.json \
        $CI_REGISTRY_IMAGE:$CI_COMMIT_SHA
 
    - echo "✓ Trivy scan complete"
 
  artifacts:
    when: always      # save reports even when scan finds issues
    expire_in: 30d
    paths:
      - trivy-report.json
      - gl-container-scanning-report.json
    reports:
      # Populates GitLab → Security → Container Scanning dashboard
      container_scanning: gl-container-scanning-report.json
 
  allow_failure: false  # CRITICAL findings block the deploy stage
 
  only:
    - main
    - develop

# ─── Deploy to K3s via ArgoCD ─────────────────────────────────────────────────
deploy:
  stage: deploy
  image: alpine/k8s:1.32.11
  before_script:
    - echo "Setting up kubectl for K3s cluster..."

    - |
      if [ -z "$KUBE_CONFIG_BASE64" ]; then
        echo "ERROR: KUBE_CONFIG_BASE64 is not set!"
        echo "Please add it to GitLab CI/CD Variables (Settings → CI/CD → Variables)"
        exit 1
      fi

    - mkdir -p ~/.kube
    - echo "$KUBE_CONFIG_BASE64" | base64 -d > ~/.kube/config
    - chmod 600 ~/.kube/config

    - |
      if [ ! -s ~/.kube/config ]; then
        echo "ERROR: Kubeconfig is empty after decoding!"
        exit 1
      fi

    - echo "✓ Kubeconfig decoded successfully"
    - kubectl version --client
    - kubectl cluster-info
    - kubectl get nodes

  script:
    - echo "Deploying NextJS frontend to Kubernetes..."

    # Create frontend namespace
    - kubectl create namespace kyc-blockchain-frontend --dry-run=client -o yaml | kubectl apply -f -

    # Create GitLab registry pull secret in frontend namespace
    - |
      kubectl create secret docker-registry gitlab-registry \
        --docker-server=$CI_REGISTRY \
        --docker-username=$CI_REGISTRY_USER \
        --docker-password=$CI_REGISTRY_PASSWORD \
        --docker-email=$GITLAB_USER_EMAIL \
        -n kyc-blockchain-frontend \
        --dry-run=client -o yaml | kubectl apply -f -

    # Apply ArgoCD application manifest
    - echo "Applying frontend ArgoCD application..."
    - kubectl apply -f k8s/argocd/

    # Wait for ArgoCD to register the application
    - sleep 5

    # Trigger ArgoCD sync
    - |
      if kubectl get application kyc-blockchain-frontend -n argocd >/dev/null 2>&1; then
        echo "Syncing kyc-blockchain-frontend application..."
        kubectl patch application kyc-blockchain-frontend -n argocd \
          --type merge \
          -p '{"operation":{"sync":{"syncStrategy":{"apply":{"force":true}}}}}' || true
      else
        echo "kyc-blockchain-frontend ArgoCD application not found, skipping sync"
      fi

  when: manual
  only:
    - main
    - develop
